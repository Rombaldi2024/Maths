{\Large\textbf{Corrigé de l'épreuve d'analyse}}\\[.3cm]%
\textit{Généralisation du théorème de Hahn-Banach; théorème de John.}\\[.3cm]
\hspace*{5cm}\textbf{Solution par F. SUFFRIN}\\[.3cm]
\begin{center}
\textbf{PREMI\`ERE PARTIE : Etude d'exemples\\
A.} \textit{Cas des espaces euclidiens}\\[.6cm]\end{center}
\textbf{\bm 1 $F$ est isométrique à $l_n^2$ :} Désignons par $(\xi_1,\cdots,\xi_n)$ une base
orthonormale de $F$. L'application $$\xi=\sum_{i=1}^nx_i\xi_i\lto(x_1,\cdots,x_n)$$ réalise une
isométrie de $F$ dans $l_n^2$, d'où le résultat.\\[.3cm]
\textbf{\bm 2-a $\|p\|=1$ ssi $E\perp E_1$ :} Montrons plus généralement que $\|p\|\ie1$ si et seulement si $E\perp
E_1$, résultat qui va nous servir par la suite.\\ Le théorème de Pythagore montre que la condition
est suffisante.\\ Supposons réciproquement $\|p\|\ie1$ et considérons alors $(x,y)\in E\times E_1$.
On a pour tout $t$ réel $$\|p(tx+y)\|=\|tx\|\ie\|tx+y\|,$$ d'où par élévation au carré  : $$\qqs
t\in\R,2t<x,y>+\|y\|^2\se0.$$ Les seules fonctions affines sur $\R$  qui gardent un signe constant
 sont les applications constantes. On en déduit $<x,y>=0$ et donc
 $E\perp E_1$.\\[.3cm]
\textbf{\bm 2-b $\pi(E,F)=1$ :} $F$ étant de dimension finie, $\pi(E,F)=\ds\inf_{E_1}\|p_{E,E_1}\|$ où
$E_1$ parcourt l'ensemble des sous espaces supplémentaires de $E$ dans $F$. Ce qui précède conduit
alors à $$\pi(E,F)=\|p_{E,E^\perp}\|=1.$$\\
\begin{center}
\textbf{B.} \textit{Exemples dans $l_n^\infty$}\\[.6cm]\end{center}
\textbf{\bm 3 Expression de $\|u\|$ :} Introduisons l'entier $i_0$ de $[1,n]$ tel que
$$\sum_{j=1}^n|a_{i_0j}|=\max_{1\ie i\ie n}\sum_{j=1}^n|a_{ij}|.$$ Pour tout
$x=\ds\sum_{i=1}^nx_ie_i$, on a
$$\|u(x)\|_\infty=\left\|\sum_{i=1}^n(\sum_{j=1}^na_{ij}x_j)e_i\right\|_\infty=\max_{1\ie i\ie
n}\left|\sum_{j=1}^na_{ij}x_j\right|.$$ Comme pour tout $1\ie i\ie n$
\begin{align}
\left|\sum_{j=1}^na_{ij}x_j\right|&\nonumber\ie\sum_{j=1}^n|a_{ij}||x_j|\\
&\nonumber\ie\|x\|_\infty\sum_{j=1}^n|a_{ij}|\\ &\nonumber\ie\|x\|_\infty\max_{1\ie k\ie n}
\sum_{j=1}^n|a_{kj}|,
\end{align}
on en déduit $$\|u(x)\|_\infty\ie\|x\|_\infty\max_{1\ie k\ie n} \sum_{j=1}^n|a_{kj}|,$$ pour tout
$x\in\R^n$ avec égalité pour $\ds x=\sum_{j=1}^n\textnormal{sign}(a_{i_0j})e_j$, où pour tout
$t\in\R$ $$\textnormal{sign}(t)=\left\{\begin{array}{l}-1\textnormal{ si } t<0\\ 1\textnormal{ si
}t\se0\end{array}\right..$$ En conclusion l'expression demandée est $$\|u\|=\max_{1\ie i\ie n}
\sum_{j=1}^n|a_{ij}|.$$
\textbf{\bm 4-a $\|p_{H,D}\|=\max(\varphi(\alpha_1),\varphi(\alpha_2),\varphi(\alpha_3))$ :} $p$ a
pour matrice dans $e$ $$\left(\begin{array}{lll}1-\alpha_1&-\alpha_1&-\alpha_1\\
-\alpha_2&1-\alpha_2&-\alpha_2\\-\alpha_3&-\alpha_3&1-\alpha_3\\\end{array}\right)$$ d'où
$$\|p_{H,D}\|=\max(\varphi(\alpha_1),\varphi(\alpha_2),\varphi(\alpha_3)).$$
\textbf{\bm 4-b $\|p_{H,D}\|\se4/3$ :} Pour tout $t\in[0,1]$ et $a<b$
\begin{align}
\varphi(ta+(1-t)b)&\nonumber=|t(1-a)+(1-t)(1-b)|+2|ta+(1-t)b|\\ &\nonumber\ie
t|1-a|+(1-t)|1-b|+2t|a|+2(1-t)|b|=t\varphi(a)+(1-t)\varphi(b),
\end{align}
d'où on déduit la convexité de $\varphi$.\\ De ce fait on peut écrire $$\varphi({1\over3})
\ie{1\over3}(\varphi(\alpha_1)+\varphi(\alpha_2)+\varphi(\alpha_3))\ie\max(\varphi(\alpha_1),\varphi(\alpha_2),\varphi(\alpha_3))$$
et donc $\|p_{H,D}\|\se4/3.$\\[.3cm]
\textbf{\bm Etude du cas d'égalité :} Le cas d'égalité  conduit à
$${4\over3}=\varphi(\alpha_1)=\varphi(\alpha_2)=\varphi(\alpha_3)=\max(\varphi(\alpha_1),\varphi(\alpha_2),\varphi(\alpha_3))$$
et en particulier $$\{\alpha_1,\alpha_2,\alpha_3\}\subset\{-{1\over9},{1\over3}\}$$ comme le montre
une étude sommaire de la fonction $\varphi$.\\
 La condition
$\alpha_1+\alpha_2+\alpha_3=1$ donne alors $\alpha_1=\alpha_2=\alpha_3=1/3$.

En résumé seule la droite $\cal D$ d'équation $x_1=x_2=x_3$ réalise le cas d'égalité $\|p_{H,\cal
D}\|={4/3}$.\\[.3cm]
\textbf{\bm $\pi(H,l_3^\infty)=4/3$ :} Tout supplémentaire de $H$ est dirigé par un vecteur
 de la forme $\alpha_1e_1+\alpha_2e_2+\alpha_3e_3$, avec $\alpha_1+\alpha_2+\alpha_3=1$. Le résultat découle de
 l'étude qui précède.\\[.3cm]
\textbf{\bm 4-c Calcul de $\ds\sigma(H,l_3^\infty)$ :} Pour tout supplémentaire $D$ de $H$, on a
$$\|s_{H,D}\|=\|2p_{H,D}-Id_{l_3^\infty}\|\se2\|p_{H,D}\|-1\se{5\over3}$$ avec cas d'égalité pour
$D=\cal D$. On en déduit $\sigma(H,l_3^\infty)=5/3.$\\[.3cm]
\textbf{\bm 5 Calcul de $\pi(H_n,l_n^\infty)$ :} Considérons $D$ un supplémentaire de $H$ dirigé par
le vecteur $\ds\sum_{i=1}^n\alpha_ie_i$ avec $\ds\sum_{i=1}^n\alpha_i=1$. La matrice de $p_{H,D}$
dans la base $e$ est $(-\alpha_i+\delta_{ij})$ d'où on déduit $\|p_{H,D}\|=\max_{1\ie i\ie
n}\Psi(\alpha_i),$ où l'on a posé pour tout réel $t$, $$\Psi(t)=|1-t|+(n-1)|t|.$$ On calque alors
le raisonnement qui précède : \\ $\Psi$ est convexe, ce qui assure $$\Psi({1\over n})\ie{1\over
n}\sum_{i=1}^n\Psi(\alpha_i)\ie\|p_{H,D}\|$$ c'est à dire $$\|p_{H,D}\|\se{2(n-1)\over n}$$ avec
pour cas d'égalité  la droite d'équation $x_1=x_2=\cdots=x_n$.

En résumé on a la relation $$\pi(H_n,l_n^\infty)={2(n-1)\over n}.$$
\textbf{\bm Calcul de $\sigma(H_n,l_n^\infty)$ :} Le raisonnement du \textbf{I-4-c} peut être
reproduit. On établit alors la relation $$\sigma(H_n,l_n^\infty)={3n-4\over n}.$$
\begin{center}
\textbf{DEUXI\`EME PARTIE : Conséquences du théorème de Hahn-Banach\\
A.} \textit{Encadrements de $\pi(E,F)$ et de $\sigma(E,F)$}\\[.6cm]\end{center}
\textbf{\bm 1-a Existence du maximum sur $(B_E)^n$ :} $E$ étant de dimension finie, $\psi$ est
continue et $(B_E)^n$ est compact. Par suite $\psi$ est bornée sur $(B_E)^n$, atteint ses bornes et
donc admet un maximum.\\[.3cm]
\textbf{\bm 1-b $e$ est libre et ses vecteurs sont de norme 1 :} Considérons $(\xi_1,\cdots,\xi_n)$ une base de $E$
constituée de vecteurs de $B_E$. $\psi$ étant multilinéaire alternée non nulle, on a
$$\psi(\xi_1,\cdots,\xi_n)\psi(-\xi_1,\cdots,\xi_n)<0$$ et par suite $\Psi(e)>0$. On en déduit que
 $e$ est libre.

  Par ailleurs $({1\over\|e_i\|} e_i)_{1\ie i\ie n}$ est dans $(B_E)^n$ d'où on peut écrire
 $$\psi({1\over\|e_1\|} e_1,\cdots,{1\over\|e_n\|} e_n)\ie\psi(e)$$ c'est à dire
 $$\psi(e)(\prod_{i=1}^n\|e_i\|-1)\se0.$$
Le résultat en est une conséquence immédiate.\\[.3cm]
\textbf{Remarque :} $\psi(e)$ est en fait la norme de $\psi$.\\[.3cm]
 \textbf{\bm $\|e_i^*\|=1$ :} Fixons un entier $i$ dans $[1,n]$. On a pour tout
 $x=\ds\sum_{j=1}^nx_je_j$ non nul
 $$\psi(e_1,\cdots,e_{i-1},{1\over\|x\|}x,e_{i+1},\cdots,e_n)\ie\psi(e),$$ c'est à dire
$$\psi(e_1,\cdots,e_{i-1},x,e_{i+1},\cdots,e_n)\ie\psi(e)\|x\|,$$ inégalité encore valable pour
$x=0$. En changeant $x$ en $-x$, il vient : $$\qqs x\in E,
|\psi(e_1,\cdots,e_{i-1},x,e_{i+1},\cdots,e_n)|\ie\psi(e)\|x\|.$$ Comme $\psi(e)>0$ et
$$\psi(e_1,\cdots,e_{i-1},x,e_{i+1},\cdots,e_n)=\psi(e)x_i=\psi(e)e_i^*(x),$$ on  a : $$\qqs x\in
E,|e_i^*(x)|\ie\|x\|$$ avec égalité pour $x=e_i$.

Il en résulte $\|e_i^*\|=1$, pour tout entier $i\in[1,n]$.\\[.3cm]
\textbf{\bm 1-c $E$ est de type ${\cal P}_n$ :} Considérons $H$ un espace vectoriel normé, $G$ un
sous espace vectoriel de $H$ et $u\in L(G,E)$.

En gardant les notations de \textbf{II-1-b}, introduisons les formes linéaires sur $G$
$$l_i=e_i^*\circ u$$ de telle façon à avoir en particulier : $$\qqs x\in
G,u(x)=\sum_{i=1}^nl_i(x)e_i.$$ Elles sont continues par théorèmes généraux et donc le théorème de
Hahn-Banach assure l'existence de $m_1,\cdots,m_n$ des formes linéaires continues sur $H$ qui
vérifient $m_i|_G=l_i$ et $\|m_i\|=\|l_i\|$, pour tout entier $i\in[1,n]$.

Définissons alors $v$ par : $$\qqs x\in G,v(x)=\sum_{i=1}^nm_i(x)e_i.$$ $v$ est un prolongement
continue de $u$ sur $H$ et  :
 $$\qqs x\in B_H,\|v(x)\|\ie\sum_{i=1}^n|m_i(x)|\ie\sum_{i=1}^n\|m_i\|$$
ce qui conduit à $$\|v\|\ie\sum_{i=1}^n\|l_i\|.$$ Comme $\|l_i\|\ie\|e_i^*\|\|u\|=\|u\|$ pour tout
entier $i\in[1,n]$ , on en déduit $$\|v\|\ie n\|u\|.$$ \textbf{\bm  2-a Existence de $x^*$ :} On
applique le théorème de Hahn-Banach à $l$ la forme linéaire sur $\R.x$  définie par : $$\qqs
t\in\R,l(tx)=t\|x\|.$$
 \textbf{2-b $\chi$ est un plongement :} Pour tout $x$ dans $E$, $\|\chi(x)\|=\ds\sup_{l\in
 B_{E'}}|l(x)|$ et puisque :
 $$\qqs l\in B_{E'},|l(x)|\ie\|l\|\|x\|\ie\|x\|$$ avec égalité pour $l=x^*$, il en résulte :
 $$\qqs x\in E,\|\chi(x)\|=\|x\|.$$
 \textbf{\bm 2-c $\|^tu\|=\|u\|$ :} $^tu$ est linéaire de $F'$ dans $E'$ par théorèmes généraux et :
 $$\qqs l\in B_{F'},\|^tu(l)\|\ie\|l\|\|u\|\ie\|u\|.$$
On en déduit  $^tu\in L(F',E')$ avec $\|^tu\|\ie\|u\|$.

Par ailleurs fixons un réel $\varepsilon>0$; il existe $x\in B_E$ tel que
$\|u(x)\|\se\|u\|-\varepsilon$ puis d'après \textbf{II-2-a} il existe $l\in B_{F'}$ telle que
$l(u(x))=\|u(x)\|$. Il en résulte $$\|u\|-\varepsilon\ie l(u(x))\ie \|l\circ u\|$$ c'est à dire
$$\|u\|-\varepsilon\ie\|^tu(l)\|\ie\|^tu\|.$$ On en déduit $\|u\|\ie\|^tu\|$ et l'égalité
annoncée.\\[.3cm]
\textbf{\bm 3-a Propriété de $s\lto-^ts$ :} Pour tout supplémentaire $E_1$ de $E$ dans $F$, on vérifie la relation $$-^ts_{E,E_1}=s_{E^\perp,E_1^\perp}.$$
On en déduit que l'application proposée établit une bijection entre les symétries de $F$ par
rapport à $E$ et les symétries de $F'$ par rapport à $E^\perp$ avec pour application réciproque :
$$s_{E^\perp,H}\lto s_{E, H^\perp},$$  pour tout supplémentaire $H$ de $E^{\perp}$ dans $F'$,
l'identification de $F$ à son bidual étant permise puisqu'en dimension finie $\chi$ est une
isométrie.\\[.3cm]
\textbf{\bm 3-b $\pi(E,F)-1\ie\pi(E^\perp,F')\ie\pi(E,F)+1$ :} $F$ étant de dimension finie,
$\pi(E,F)$ et $\pi(E^\perp,F')$ sont réels. Avec des notations évidentes pour les bornes
inférieures, on a successivement
$$\pi(E^\perp,F')=\inf_H\|p_{E^\perp,H}\|=\inf_H\|{1\over2}(s_{E^\perp,H}+Id_{F'})\|.$$  La
bijection précédente permet alors d'écrire
$$\pi(E^\perp,F')=\inf_{E_1}\|{1\over2}(-^ts_{E,E_1}+Id_{F'})\|=\inf_{E_1}\|{1\over2}(-s_{E,E_1}+Id_{F})\|,$$
la dernière égalité découlant de \textbf{II-2-c}, et par suite
$$\pi(E^\perp,F')=\inf_{E_1}\|Id_F-p_{E,E_1}\|.$$ Le passage à la borne inférieure sur les
supplémentaires de $E$ dans $F$ dans les inégalités
$$\|p_{E,E_1}\|-1\ie\|Id_F-p_{E,E_1}\|\ie\|p_{E,E_1}\|+1$$ conduit alors à
$$\pi(E,F)-1\ie\pi(E^\perp,F')\ie\pi(E,F)+1.$$
\textbf{\bm 3-c $\sigma(E,F)\ie3$ si $E$ est un hyperplan :} Considérons $m$ un vecteur directeur
unitaire de $E^\perp$.

 $F$ étant de dimension finie, $$\|m\|=\ds\sup_{x\in B_F}|m(x)|=\max_{x\in B_F}|m(x)|$$ et par suite
 il existe $\xi\in B_F$ tel que $m(\xi)=1$.\\La droite $D=\R.\xi$ est un supplémentaire de $E$
 dans $F$ et :
 $$\qqs x\in F,s_{E,D}(x)=x-2m(x)\xi.$$
Il en résulte : $$\qqs x\in B_F,\|s_{E,D}(x)\|\ie\|x\|+2|m(x)|\|\xi\|\ie3$$ ou encore
$\|s_{E,D}\|\ie3$, c'est à dire $\sigma(E,F)\ie3$.\\[.3cm]
\textbf{\bm $\pi(E,F)\ie2$ :} En gardant les notations précédentes, il vient
$$\|p_{E,D}\|=\|{1\over2}(s_{E,D}+Id_F)\|\ie{1\over2}(\|s_{E,D}\|+1)\ie2.$$ Le résultat en
découle.\\[.3cm]
\textbf{\bm Ces inégalités sont optimales :} Les calculs effectués à la question \textbf{I-5}
montrent que ces majorations ne peuvent être améliorées.\\[.3cm]
\begin{center}
\textbf{B.} \textit{ Etude de $\omega$}\\[.6cm]\end{center}
\textbf{\bm 4 $\pi(E,F)\ie\lambda$ :} $E$ étant de type ${\cal P}_\lambda$, il existe $v$ continue
linéaire de $F$ dans $E$ qui prolonge $Id_E$ et tel que  $\|v\|\ie\lambda\|Id\|=\lambda$. $v$ est
alors un projecteur sur $E$, d'où en déduit $\pi(E,F)\ie\lambda$.\\[.3cm]
\textbf{\bm 5-a-b ${\cal B}(X,\R)$ est de type ${\cal P}_1$ :} Le cas $l_n^\infty$ correspond
à $X=\{1,\cdots,n\}$.

Considérons $H$ un espace vectoriel normé, $G$ un sous espace de $H$ et $u$ une application
linéaire continue  de $G$ dans $F={\cal B}(X,\R)$.\\Pour $t$ dans $X$, notons $l_t$ la forme
linéaire définie sur $F$ par : $$\qqs f\in F,l_t(f)=f(t).$$ L'inégalité $$\qqs f\in
F,|l_t(f)|\ie\|f\|_\infty$$ fournit  $l_t\in B_{F'}$ et par suite $l_t\circ u\in G'$. D'après le
théorème de Hahn-Banach, il existe $m_t\in H'$ prolongeant $l_t\circ u$ et telle que
$\|m_t\|=\|l_t\circ u\|$.

Introduisons alors l'application $v$ qui à $x\in H$ associe $g=v(x)$, où $g$ désigne l'application
définie sur $X$ par  : $$\qqs t\in X,g(t)=m_t(x).$$ $v$ est linéaire sur $H$ et pour tout $x\in H$,
on a : $$\qqs t\in X,|v(x)(t)|\ie\|m_t\|\|x\|\ie\|u\|\|x\|;$$  $v$ prend alors ses valeurs dans $F$
et : $$\qqs x\in H,\|v(x)\|\ie\|u\|\|x\|,$$ ce qui assure $v\in L(H,F)$ avec $\|v\|\ie\|u\|$.

Par ailleurs pour tout $x\in G$, on a : $$\qqs t\in X,v(x)(t)=m_t(x)=l_t\circ u(x)=u(x)(t)$$ ou
encore $v(x)=u(x)$, c'est à dire $v$ prolonge $u$ dans $H$.

En conclusion ${\cal B}(X,\R)$ est de type ${\cal P}_1$.\\[.3cm]
 \textbf{\bm 5-c Tout evn peut être plongé dans un evn de type ${\cal P}_1$ :}
 Considérons $E$ un espace vectoriel normé et $X=B_{E'}$.

 $X$ contient la forme nulle donc est non vide, d'où l'on peut poser $F={\cal B}(X,\R)$ qui
 est de type ${\cal P}_1$. D'après \textbf{II-2-b}, $\chi$ plonge $E$ dans $E''\subset F$.

 Le résultat en découle.\\[.3cm]
 \textbf{\bm 6  Equivalence souhaitée :} Supposons (i) vérifiée. $\omega$ prend la même valeur sur des espaces vectoriels
 normés isométriques. On en déduit $\omega(E)=\omega(\chi(E))$, ce qui permet de se ramener à
 $E\subset F$ et $\chi=Id_E$.

 La question \textbf{II-4} fournit $\pi(E,F)\ie\omega(E)$ c'est à dire $\mu\ie\omega(E)$.

 Par ailleurs fixons $\varepsilon>0$; il existe $p$ un projecteur continue sur $E$ tel que
 $\|p\|\ie\pi(E,F)+\varepsilon=\mu+\varepsilon$. Considérons alors $H$ un espace vectoriel normé,
 $G$ un sous espace de $H$ et $u$ une application linéaire continue de $G$ dans $E$.
 En particulier $u\in L(G,F)$ et par suite, $F$ étant du type ${\cal P}_1$, il existe un
 prolongement continue $v_1$ de $u$ à $H$ tel que $\|v_1\|\ie\|u\|.$

 Alors $v=p\circ v_1$ appartient à $L(H,E)$ et prolonge $u$ à $H$ avec $$\|v\|\ie\| p\|\|v_1\|\ie(\mu+\varepsilon)\|u\|.$$
Il en résulte $\omega(E)\ie\mu+\varepsilon$ et par suite $\omega(E)=\mu$, ce qui constitue (ii).

La réciproque s'obtient à l'aide du plongement $\chi$ et l'espace vectoriel normé $F$ de la
question
\textbf{II-5-c}, combinés avec ce qui précède.

\begin{center}
\textbf{TROISI\`EME PARTIE : Les minorations de Sobczyk}\\[.3cm]\end{center}
\textbf{\bm 1-a $p$ et $q$ sont des projecteurs sur un même sous ev ssi $p\circ q=q$ et $q\circ
p=p$ :} La condition est clairement nécessaire.

Supposons réciproquement que $p$ et $q$ vérifient les identités ci-dessus. Alors on a
successivement $$p\circ p=p\circ(q\circ p)=(p\circ q)\circ p=q\circ p=p$$ d'où $p$ est un
projecteur de $F$. Par ailleurs, on a $\textnormal{Im }p=\textnormal{Im }q\circ
p\subset\textnormal{Im }q$.\\ $p$ et $q$ jouant des rôles symétriques, sont donc des projecteurs
sur un même sous espace vectoriel de $F$.\\[.3cm]
\textbf{\bm 1-b CNS pour que $s'$ soit une symétrie par rapport à $E$ :} $p={1\over2}(s+Id_F)$
étant un projecteur sur $E$, on déduit que $s'=s+v$ est une symétrie par rapport à $E$ si et
seulement si $p$ et $q={1\over2}(s'+Id_F)$ vérifient les conditions qui précèdent.\\ Une condition
équivalente est donc donnée par $s\circ v=v$ et $v\circ s=-v$, ce qui constitue le résultat.

En outre dans ce dernier cas, on a $$\textnormal{Tr }(v)=\textnormal{Tr }(s\circ v)=\textnormal{Tr
}(v\circ s)=-\textnormal{Tr }(v),$$ c'est à dire $v$ est de trace nulle.\\[.3cm]
\textbf{\bm 2-a Inégalités souhaitées :} La relation
$$({1\over\sqrt n}A)^2={1\over n}nI_n=I_n$$ montre que $s$ est une symétrie de $l_n^\infty$.

 Par ailleurs si on pose $S=(s_{ij})$ et $V=(v_{ij})$, on a $$(S+V)S=I_n+VS=I_n-V.$$
$S$ est symétrique donc l'interprétation de cette égalité sur le $i-$ième terme de la diagonale
donne $$1-v_{ii}=\sum_{j=1}^n(s_{ij}+v_{ij})s_{ij},$$ pour tout $i\in[1,n]$. On en déduit : $$\qqs
i\in[1,n], 1-v_{ii}\ie\sum_{j=1}^n|s_{ij}+v_{ij}||s_{ij}|\ie{1\over\sqrt
n}\sum_{j=1}^n|s_{ij}+v_{ij}|$$ et donc : $$\qqs i\in[1,n], 1-v_{ii}\ie{1\over\sqrt n}\|s+v\|.$$
\textbf{\bm 2-b $\sigma(E_A,l_n^\infty)\se\sqrt n$ :}
 Considérons $s'$ une symétrie de $l_n^\infty$ par rapport à $E_A$; l'endomorphisme  $v=s'-s$
 vérifie les conditions du \textbf{III-1-b}. Si $(v_{ij})$ désigne sa matrice dans la base canonique, les inégalités établies auparavant
 s'écrivent :
$$\qqs i\in[1,n], 1-v_{ii}\ie{1\over\sqrt n}\|s'\|.$$ $v$ étant de trace nulle, une sommation donne
$\|s'\|\se\sqrt n$.

 On en déduit $\sigma(E_A,l_n^\infty)\se\sqrt n$.\\[.3cm]
\textbf{\bm $\pi(E_A,l_n^\infty)\se{1\over2}(\sqrt n-1)$ :} Considérons $p$ un projecteur de
$l_n^\infty$ sur $E_A$ ; $2p-Id_{l_n^\infty}$ est une symétrie de $l_n^\infty$ par rapport à $E_A$
et par suite $\|2p-Id_{l_n^\infty}\|\se\sqrt n$.\\En particulier on a $2\|p\|+1\se\sqrt n,$ ou
encore $\|p\|\se{1\over2}(\sqrt n-1)$, c'est à dire $$\pi(E_A,l_n^\infty)\se{1\over2}(\sqrt n-1).$$
\textbf{\bm 3 $(A_k)\in{\cal H}(2^k)$ :} Montrons par récurrence que $A_k$ est un élément de ${\cal
H}(2^k)$, pour tout $k\in\N$.

La propriété est vraie au rang 0; supposons la vraie au rang $k\se0$, c'est à dire supposons
$A_k\in{\cal H}(2^k)$.

Alors $A_{k+1}$ est une matrice symétrique de $M_{2^{k+1}}(\R)$, dont les coefficients sont égaux à
$\pm1$. Un produit par blocs adjoint à l'hypothèse de récurrence donne alors
$(A_{k+1})^2=2^{k+1}I_{2^{k+1}}$.

Ceci constitue le résultat au rang $k+1$, donc il est vrai pour tout $k\in\N$.\\[.3cm]
\textbf{\bm dim $E_{A_k}=2^{k-1}$ :} $E_{A_0}$ est de dimension 1. Sinon pour tout entier $k\se1$,
${1\over2^{k/2}}A_k$ est la matrice représentative dans la base canonique de la symétrie par
rapport à $E_{A_k}$ et donc ${1\over2}({1\over2^{k/2}}A_k+I_{2^k})$ est la matrice représentative
dans la base canonique du projecteur associée. La dimension de $E_{A_k}$ est donc égale au rang de
ce projecteur, c'est à dire sa trace. $A_k$ étant de trace nulle, il vient alors par passage à la
trace : $$\qqs k\se1,\textnormal{dim } E_{A_k}={1\over2}\textnormal{Tr }(I_{2^k})=2^{k-1}.$$
\textbf{\bm 4-a  $E$ est fermé si $\pi(E,F)<\infty$ :} $\pi(E,F)<\infty$ donc il
existe $p$ un projecteur continue de $F$ sur $E$. Alors $E$ est fermé comme sous espace propre de
$p$.\\[.3cm]
\textbf{\bm 4-b Construction de la suite $(E_k)$ :} Pour tout $k\in\N$, l'application
$\chi_k$ de $l_{2^k}^\infty$ dans $F$ qui à $x$ associe $(a_n)$ définie par : $$\qqs
n\in\N,a_n=\left\{\begin{array}{l}x_{n-2^k+1}\textnormal{ pour }n\in[2^k,2^{k+1}[\\0\textnormal{
sinon}\end{array}\right.$$ induit une isométrie de  $l_{2^k}^\infty$ dans $F_k$. Notant
$E_k=\chi_k(E_{A_k})\subset F_k$, on a
$$\pi(E_k,F_k)=\pi(E_{A_k},l_{2^k}^\infty)\se{1\over2}(2^{k/2}-1),$$ pour tout $k\in\N$.

La suite $(E_k)$ répond à la question alors.\\[.3cm]
\textbf{\bm $\pi(E,F)=+\infty$ :} Nous allons montrer qu'il n'existe pas de projecteur continue de
$F$ sur $E$.

Considérons $p$ un projecteur de $F$ sur $E$ et un entier naturel $k$. Le projecteur $p_k$ de $F$
sur $F_k$ définie par : $$(x_n)\lto(0,\cdots,0,x_{2^n},\cdots,x_{2^{n+1}-1},0,\cdots)$$  est
continue de norme 1, pour tout $k\in\N$.

Si $x$ est un vecteur de $\ds\bigoplus_{j\se0}E_j$, $p_k(x)$ appartient à $E_k$; $p_k$ étant
continue et $E_k$ fermé puisque de dimension finie, on a : $$\qqs x\in E,p_k(x)\in E_k,$$ pour tout
$k\in\N$. On en déduit que l'application $q_k$ restriction à $F_k$ de $p_k\circ p$ constitue un
projecteur de $F_k$ sur $E_k$ et par suite vérifie $$\|q_k\|\se\pi(E_k,F_k),$$ pour tout $k\in\N$.
Si $p$ était continue, on aurait alors : $$\qqs k\in\N,\|p\|\se\|q_k\|\se{1\over2}(2^{k/2}-1),$$ ce
qui est impossible.

On en déduit $\pi(E,F)=+\infty$ et $\omega(E)=+\infty$, d'après \textbf{II-4} .

En particulier la réciproque de l'énoncé du \textbf{III-4-a} est fausse.\\[.3cm]
\begin{center}
\textbf{QUATRI\`EME PARTIE : Le théorème de John\\
A.}\textit{ Preuve du théorème}\\[.3cm]\end{center}
\textbf{\bm 1-a Intégrabilité de $x\lto e^{-q(x)}$ :} Pour $q$ dans $Q$, notons $u(q)$ l'opérateur autoadjoint de
$\R^n$  qui lui est canoniquement associé; la réduction de $u(q)$ dans une base orthonormale de
vecteurs propres donne, en désignant par $\lambda_1,\cdots,\lambda_n$ les valeurs propres associées
$$I(q)=\int_{\R^n}\exp(-\sum_{1\ie i\ie n}\lambda_ix_i^2)dx.$$ L'application est donc intégrable si
et seulement si chaque valeur propre est $>0$ c'est à dire $q\in Q_+$.

Dans ce dernier cas on a $$I(q)={\sqrt{\pi^n}\over \sqrt{det(u(q))}}.$$
\textbf{\bm 1-b $I(q)=v({\cal E}_q)\ds\int_0^{\infty}t^{n/2}e^{-t}dt$ :} Intégrons de deux façons l'intégrale double proposée. On a
$$\int\hspace{-.25cm}\int_Ce^{-t}dxdt=\int_{\R^n}\int_{q(x)}^{+\infty}e^{-t}dt\hspace{.2cm}dx=I(q).$$
Par ailleurs on a $$\int\hspace{-.25cm}\int_Ce^{-t}dxdt=\int_0^{+\infty}f(t)dt,$$ avec : $$\qqs
t\se0,f(t)=e^{-t}\int_{q(x)\ie t}dx\underset{x=\sqrt t y}{=}e^{-t}t^{n/2}v({\cal E}_q).$$ La
relation demandée en est une conséquence immédiate.\\[.3cm]
\textbf{\bm 1-c Continuité de $I$ :} L'application $u:q\lto u(q)$ est linéaire sur $Q$ donc
continue.

 La continuité de $I$ résulte donc de l'expression obtenue en \textbf{IV-1-a}.\\[.3cm]
\textbf{\bm 2-a $K$ est un convexe compact de $Q$ :} Posons pour $q$ dans $Q$
$$\|q\|_N=\sup_{x\in S_N}|q(x)|,$$ où $S_N$ désigne la sphère unité de $(\R^n,N)$.

$\|.\|_N$ est  une norme sur $Q$ et alors $K$ est convexe compact comme la trace du convexe fermé
$\overline{Q_+}$ sur la boule unité fermé de $(Q,\|.\|_N)$.

En outre pour $q$ donné dans $Q_+$, ${1\over\|q\|_N}q$ appartient à $K\cap Q_+$ qui donc est non
vide.\\[.3cm]
\textbf{\bm 2-b Existence de $q_N$ :} $I$ est continue  sur le compact $K$ donc est minorée
et atteint sa borne inférieure. Il existe alors $q_N\in K$ telle que $I(q_N)=\ds\min_{q\in K}I(q)$;
pour $q$ dans $K\cap Q_+$ on a $I(q_N)\ie I(q)<+\infty$, ce qui assure $q_N\in K\cap Q_+$.

En particulier le volume $v({\cal E}_{q_N})$ de son ellipso\"\i de est minimum.

Le résultat en découle alors.\\[.3cm]
\textbf{\bm 2-c Unicité de $q_N$ :} $K$ étant convexe, l'élément $q+q_N\over2$ est dans $K$ et
alors $I(q_N)\ie I({q+q_N\over2})$.

 Par ailleurs l'inégalité de Cauchy-Schwarz donne
$$I\left({q+q_N\over2}\right)=\int_{\R^n}e^{-q(x)\over2}e^{-q_N(x)\over2}dx\ie
\left(\int_{\R^n}e^{-q(x)}dx\int_{\R^n}e^{-q_N(x)}dx\right)^{1\over2}$$ ou encore $$I(q_N)\ie
I\left({q+q_N\over2}\right)\ie\sqrt{I(q)I(q_N)}=I(q_N)$$ c'est à dire
$$I\left({q+q_N\over2}\right)=I(q_N),$$ qui correspond au cas d'égalité. Les fonctions $x\lto
e^{-q(x)\over2}$ et $e^{-q_N(x)\over2}$ sont alors proportionnelles; comme elles prennent la valeur
1 en 0 elles sont égales, c'est à dire $q=q_N$.\\[.3cm]
\textbf{\bm 3-a Existence de $a$ :} $N'=\sqrt{q_N}$ est une norme sur $\R^n$ et sa sphère unité associée est
$$S_{N'}=\{x\in\R^n;q_N(x)=1\}.$$ $N$ est continue sur $\R^n$ du fait de l'équivalence des normes.
Elle est donc majorée sur le compact $S_{N'}$ et atteint sa borne supérieure.

Un élément $a\in S_{N'}$ en lequel cette borne est atteinte répond à la question.\\[.3cm]
\textbf{\bm 3-b Convexité de $\varphi$ et inégalité souhaitée :} $\varphi$ est convexe comme
composée d'une application affine et de $N$, qui est convexe.

Par ailleurs la définition de $a$ assure : $$\qqs x\in\R^n,N(x)\ie N(a)N'(x).$$  $a$ et $y$ étant
orthogonale, on a donc : $$\qqs t\in\R,\varphi(t)\ie N(a)\sqrt{1+t^2q_N(y)}.$$
\textbf{\bm Etude du minimum de $\varphi$ :} Il s'agit de montrer :
$$\qqs t\in\R,N(a)\ie N(a+ty).$$ L'inégalité est vraie à l'origine; sinon fixons un réel $t>0$.

Alors pour tout réel $s\se0$, on a \begin{align} N(a)&\nonumber\ie{t\over s+t}\varphi(-s)+{s\over
s+t}\varphi(t)\\&\nonumber\ie {t\over s+t} N(a)\sqrt{1+s^2q_N(y)}+{s\over
s+t}\varphi(t)=g(s).\end{align} $g$ est dérivable sur $[0,+\infty[$ et admet un minimum à
l'origine, d'où $g'(0)\se0$  et donc $\varphi(t)\se N(a)$.

Le résultat pour $t<0$ s'obtient en remplaçant $y$ par $-y$ dans  le raisonnement précédent.

En résumé $\varphi$ présente un minimum à l'origine.\\[.3cm]
\textbf{\bm $N(\pi_a(x))\ie N(x)$ :} L'inégalité est vraie pour $x$ dans $H$.

 Sinon pour $\ds x={1\over
t}a+y$ avec $(t,y)\in\R^*\times H$, on a $\ds\pi_a(x)={1\over t}a$ et l'inégalité $N(a)\ie N(a+ty)$
donne $$N(\pi_a(x))\ie N(x).$$
\textbf{\bm 3-c $q\in Q_+$ :} $q$ est une forme quadratique positive sur $\R^n$ puis un vecteur où elle
s'annule a ses composantes nulles sur $\R.a$ et $H$, d'où $q$ est un élément de $Q_+$.\\[.3cm]
\textbf{\bm $I(q)=(1+\varepsilon)^{-1/2}(1-\delta)^{-{(n-1)/2}}I(q_N)$ :}  Dans une base orthonormale
 de $\R^n$ telle que le premier vecteur soit $a$, la matrice de $q$ est
 $$\textnormal{Diag}(1+\varepsilon,1-\delta,\cdots,1-\delta).$$
On en déduit  $$det(u(q))=(1+\varepsilon)(1-\delta)^{n-1}det(u(q_N)).$$ Le résultat est une
conséquence immédiate de la relation établie en \textbf{IV-1-a}.\\[.3cm]
\textbf{\bm Encadrement demandé :} Le théorème de Pythagore donne pour tout $x$ dans $\R^n$,
$$q_N(\pi_a(x))+q_N(\pi_H(x))=q_N(x)$$ qui conduit à
$$q(x)=(1-\delta)q_N(x)+(\varepsilon+\delta)q_N(\pi_a(x))\ie(1-\delta)N^2(x)+(\varepsilon+\delta)q_N(\pi_a(x))$$
c'est à dire $$q(x)\ie N^2(x)-\left[\delta N^2(x)-(\varepsilon+\delta)q_N(\pi_a(x))\right].$$ Par
ailleurs l'autre inégalité équivaut à établir : $$\qqs x\in\R^n,N^2(a)q_N(\pi_a(x))\ie N^2(x).$$ On
a égalité sur $\R.a$ et donc : $$\qqs x\in\R^n,N^2(a)q_N(\pi_a(x))=N^2(\pi_a(x)).$$ L'inégalité du
\textbf{IV-3-b} donne alors : $$\qqs x\in\R^n,N^2(a)q_N(\pi_a(x))\ie N^2(x),$$
ce qui constitue le résultat. \\[.3cm]
\textbf{\bm 3-d $N(a)\ie\sqrt n$ :} Raisonnons par l'absurde; supposons $N^2(a)>n$. La fonction
$$f:x\lto(1+x)\left(1-{x\over N^2(a)-1}\right)^{n-1}$$ est dérivable sur $[0,1[$ avec $f(0)=1$ et
$$f'(0)={N^2(a)-n\over N^2(a)-1}>0.$$ Elle prend donc des valeurs $>1$ sur un voisinage de 0 dans
$]0,1[$ .\\ Choisissons $\varepsilon$ dans ce voisinage tel que $0<\varepsilon<N^2(a)-1$ et posons
$\delta={\varepsilon\over N^2(a)-1}$.\\ L'encadrement de la question précédente appliquée à la
forme quadratique $q$ qui leur est associée assure $q\in K$; mais alors l'inégalité
$$I(q)={I(q_N)\over\sqrt{f(\varepsilon)}}<I(q_N)$$ contredit la définition de $q_N$. Donc on a
$N(a)\ie\sqrt n$.


En conclusion $q_N$ est telle que le volume $v({\cal E}_{q_N})$ de son ellipso\"\i de soit minimum
et : $$\qqs x\in\R^n,N^2(x)\ie N^2(a) q_N(x)\ie nq_N(x).$$ Ceci constitue le théorème de
John.\\[.3cm]
\textbf{\bm 4  $\sigma(E,F)\ie \sqrt n$ :} $\R^n$ peut être muni d'une norme pour laquelle il est isométrique
à $F$. Il nous suffit alors de traiter le cas $F=\R^n$.

Considérons $E$ un sous espace de $\R^n$ muni d'une norme $N$. Introduisons alors la forme
quadratique $q_N$ qui lui associée dans le théorème de John et $s$ la symétrie $q_N$-orthogonale
par rapport à $E$.

Pour tout $x$ dans $\R^n$, on a $q_N(s(x))=q_N(x)$ d'où on en déduit : $$\qqs x\in\R^n,N^2(s(x))\ie
nq_N(x)\ie nN^2(x)$$ c'est à dire $\|s\|\ie\sqrt n.$

 Il en résulte $\sigma(E,\R^n)\ie\sqrt n$, d'où le résultat annoncé.\\[.3cm]
\textbf{\bm $\pi(E,F)\ie{\sqrt n+1\over2}$ :} On reprend le raisonnement qui
précède, en remarquant que le projecteur associé à la symétrie $s$ est de norme inférieure à
${\sqrt n+1\over2}$.\\[.3cm]
\textbf{\bm Liens avec les minorations de Sobczyk :} Le choix de $N=\|.\|_\infty$ donne, d'après
\textbf{III-2-b} $$\sigma(E_A,l_n^\infty)=\sqrt n\textnormal{\hspace{1cm} et\hspace{1cm} }|\pi(E_A,l_n^\infty)-{\sqrt
n\over2}|\ie{1\over2}.$$
\begin{center}
\textbf{B.} \textit{Exemples dans $l_n^\infty$}\\[.6cm]\end{center}
\textbf{\bm 5 Expression de $q_N$ :} $N$ est une fonction symétrique des composantes de $x$, invariante par changement de
 signe de l'une d'elles. Un retour à la définition
 montre alors que $q_N$ jouit de ces mêmes propriétés. Son produit scalaire associé est donc
 proportionnelle au produit scalaire usuel.

 En faisant agir l'inégalité
 $${1\over n}N^2\ie q_N\ie N^2$$ sur la base canonique et sur $e_1+\cdots+e_n$, on déduit que le coefficient de
 proportionnalité est $1/ n$.

 En conclusion on a $q\ds={1\over n}\|.\|_2^2$.\\[.3cm]
 \textbf{\bm $C=n$ est une valeur minimale :} Considérons un réel $C$ et supposons qu'un élément $q$ de $Q_+$ vérifie
 $q\ie\|.\|_\infty^2\ie Cq$, avec :
 $$\qqs x\in\R^n,q(x)=\sum_{1\ie i\ie n}a_{ii}x_i^2+2\sum_{1\ie i<j\ie n}a_{ij}x_ix_j.$$
  Alors $C\se1$ et en faisant agir l'inégalité ci-dessus sur la base canonique, on obtient :
  $$\qqs1\ie i\ie n,{1\over C}\ie a_{ii}\ie 1.$$
Par ailleurs on a $$\sum_{\varepsilon\in\{-1,1\}^n}q(\varepsilon)\ie\#\{-1,1\}^n=2^n;$$ mais le
premier terme s'écrit $$\sum_{\varepsilon\in\{-1,1\}^n}q(\varepsilon)=2^n\sum_{1\ie i\ie
n}a_{ii}+2\sum_{\varepsilon\in\{-1,1\}^n}\varepsilon_\alpha\varepsilon_\beta\sum_{1\ie i<j\ie
n}a_{ij},$$ où $\alpha$ et $\beta$ désigne deux entiers distincts de $[1,n]$; l'involution sur
$\{-1,1\}^n$ qui change la $\alpha-$ième composante en son opposée permet de déduire que la
troisième somme est nulle. On peut alors écrire $$2^n\sum_{1\ie i\ie n}a_{ii}\ie2^n$$ et par suite
$${n\over C}\ie\sum_{1\ie i\ie n}a_{ii}\ie1$$
 d'où on a l'inégalité $C\se n$.\\[.3cm]
 \textbf{\bm 6-a $\qqs g\in G,|$det $(g)|=1$ :} L'application déterminant est un morphisme de
 groupes continue de $(GL(\R^n),\circ)$ dans $(\R^*,\times)$. L'image de $G$ est alors un sous
 groupe compact de $(\R^*,\times)$, c'est à dire est inclus dans $\{-1,1\}$.

 La propriété en résulte.\\[.3cm]
\textbf{\bm 6-b $G\subset O(q_N)$ :} $N$ est une norme sur $\R^n$ du fait de la compacité de $G$.

Posons pour $g$ dans $G$, $q=q_N\circ g$.\\ $q$ est un élément de $Q_+$ qui vérifie : $$\qqs
x\in\R^n,q(x)\ie N^2(g(x)).$$ Mais la translation $h\lto h\circ g$ étant une permutation de $G$, on
a : $$\qqs x\in\R^n,N(g(x))=\sup_{h\in G}\|h\circ g(x)\|_\infty=\sup_{h\in
G}\|h(x)\|_\infty=N(x),$$ c'est à dire $q$ est un élément de $K\cap Q_+$.

Par ailleurs le changement de variable $x=g^{-1}(\xi)$ dans $I(q)$ donne
$$I(q)=|\textnormal{det}(g^{-1})|\int_{\R^n}e^{-q_N(\xi)}d\xi=\int_{\R^n}e^{-q_N(\xi)}d\xi=I(q_N).$$
L'unicité de $q_N$ conduit alors à $q=q_N$, c'est à dire $g$ est orthogonal pour  $q_N$.

En conclusion $G$ est inclus dans le groupe orthogonal de $q_N$.\\[.3cm]
\textbf{\bm 7 Autre formulation du théorème de John :} Le théorème de John assure l'existence d'un
unique élément $q$ de $Q_+$ tel que $q\ie\|.\|^2$ et tel que le volume de son ellipso\"\i de
associé $\cal E$ soit maximum.\\ De plus on a $q\ie\|.\|^2\ie nq$, ce qui équivaut à ${\cal
E}\subset B\subset\sqrt n{\cal E}$.
